启动命令#

docker run -dit \
-v /opt/jd/scripts:/jd/scripts \
-v /opt/jd/config:/jd/config \
-v /opt/jd/log:/jd/log \
-p 5678:5678 \
-e ENABLE_HANGUP=false \
-e ENABLE_WEB_PANEL=true \
--name jd \
--network bridge \
--hostname jd \
--restart always \
supermanito/helloworld
